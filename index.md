```
├── raw/                # 原始素材库（AI只读不写）
│   ├── articles/       # 网页文章
│   ├── papers/         # PDF论文
│   └── assets/         # 图片附件
├── wiki/               # AI生成的知识库
│   ├── concepts/       # 概念卡片
│   ├── entities/       # 实体介绍
│   ├── sources/        # 素材摘要
│   └── comparisons/    # 对比分析
├── CLAUDE.md           # AI操作手册
└── index.md            # 知识库导航（本文件）
```
